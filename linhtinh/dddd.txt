    private void btnAddActionPerformed(java.awt.event.ActionEvent evt) {                                       

        Connector con = new Connector();
        try {
            Connection myCon = con.OpenConnect();
            Statement statement = myCon.createStatement();
            ResultSet rs = statement.executeQuery("SELECT MAX(ID) from StudentInfo;");
            while (rs.next()) {
                stuID = rs.getInt(1);
                stuID++;
            }
        } catch (SQLException e) {
            e.printStackTrace();
        }
        txtID.setText(Integer.toString(stuID));
        btnAdd.setEnabled(false);
        btnSave.setEnabled(true);

    }                                      

    private void btnSaveActionPerformed(java.awt.event.ActionEvent evt) {                                        

        Connector con = new Connector();
        try {
            Connection myCon = con.OpenConnect();
            String insertQuery = "insert into StudentInfo (ID, [Name],Class) values(?,?,?)";
            PreparedStatement preSta = myCon.prepareStatement(insertQuery);
            preSta.setInt(1, (int) Integer.valueOf(txtID.getText()));
            preSta.setString(2, stuName);
            preSta.setString(3, txtClass.getText());
            preSta.executeUpdate();
            JOptionPane.showMessageDialog(null, "Student be was Insert");
        } catch (SQLException e) {
            e.printStackTrace();
        }
        btnSave.setEnabled(false);
        btnAdd.setEnabled(true);
        txtID.setText("");
        txtName.setText("");
        txtClass.setText("");


    }

    private void btnSearchActionPerformed(java.awt.event.ActionEvent evt) {                                          
        Connector con = new Connector();
        try {
            Connection myCon = con.OpenConnect();
            Statement statement = myCon.createStatement();
            ResultSet rs = statement.executeQuery(" SELECT * FROM StudentInfo WHERE id=( SELECT min(id) FROM StudentInfo WHERE [Name] = '"
                    + stuName
                    + "') and  [Name] = '"
                    + stuName
                    + "'");
            if (rs.next()) {
                txtID.setText(String.valueOf(rs.getInt(1)));
                txtClass.setText(rs.getString(3));
            }
            btnDelete.setEnabled(true);
            btnEdit.setEnabled(true);

        } catch (SQLException e) {
            e.printStackTrace();
        }
    }                                         

    private void txtNameKeyReleased(java.awt.event.KeyEvent evt) {                                    
        stuName = txtName.getText();
        if (stuName == null || txtName.getText().equals("")) {
            btnSearch.setEnabled(false);
        } else
            btnSearch.setEnabled(true);
    }
